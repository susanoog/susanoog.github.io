830 template<class T>
831 const char* num2binstr(T num)
832 {
833  size_t bit_length = sizeof(T) * 8;
834  static std::string str = "";
835  str = "";
836  T bit = T(1) << (bit_length-1);
837  for (size_t i=0 ;i<bit_length; i++) {
838  if (i % 8 == 0) str += " ";
839  if (num & bit) str += "1";
840  else str += "0";
841 
842  bit >>= 1;
843  }
844  return str.c_str();
845 }
