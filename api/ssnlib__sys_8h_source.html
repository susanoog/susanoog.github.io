88 
89  System_interface(int argc, char** argv)
90  {
91  /*
92  * Boot DPDK System.
93  */
94  kernel_log(SYSTEM, "[+] Booting ...\n");
95  ssnlib::print_message();
96 
97  /*
98  * DPDK init
99  */
100  int ret = rte_eal_init(argc, argv);
101  if (ret < 0) {
102  throw slankdev::exception("rte_eal_init");
103  }
104 
105  kernel_log(SYSTEM, "configure \n");
106 
107  size_t nb_cores = rte_lcore_count();
108  for (size_t lcore_id=0; lcore_id<nb_cores; lcore_id++) cpus.emplace_back(lcore_id);
109 
110  size_t nb_ports = rte_eth_dev_count();
111  for (size_t pid=0; pid<nb_ports; pid++) ports.emplace_back(pid);
112 
113  for (size_t i=0; i<RTE_CPUFLAG_NUMFLAGS; i++)
114  cpuflags[i] = rte_cpu_get_flag_name(rte_cpu_flag_t(i));
115 
116  for (auto& port : ports) port.boot();
117 
118  kernel_log(SYSTEM, "[+] DPDK boot Done! \n");
119  }
120  virtual ~System_interface() { rte_eal_mp_wait_lcore(); }
121  virtual void halt()
122  {
123  kernel_log(SYSTEM, "[+] System Halt ...\n");
124  rte_exit(0, "Bye...\n");
125  }
126  virtual void wait_all()
127  {
128  sleep(1);
129  rte_eal_mp_wait_lcore();
130  }
131 };
