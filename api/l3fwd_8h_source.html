122 /* Send burst of packets on an output interface */
123 static inline int
124 send_burst(struct lcore_conf *qconf, uint16_t n, uint8_t port)
125 {
126  struct rte_mbuf **m_table;
127  int ret;
128  uint16_t queueid;
129 
130  queueid = qconf->tx_queue_id[port];
131  m_table = (struct rte_mbuf **)qconf->tx_mbufs[port].m_table;
132 
133  ret = rte_eth_tx_burst(port, queueid, m_table, n);
134  if (unlikely(ret < n)) {
135  do {
136  rte_pktmbuf_free(m_table[ret]);
137  } while (++ret < n);
138  }
139 
140  return 0;
141 }

143 /* Enqueue a single packet, and send burst if queue is filled */
144 static inline int
145 dataplane_queue_single_packet(struct lcore_conf *qconf,
146  struct rte_mbuf *m, uint8_t port)
147 {
148  uint16_t len;
149 
150  len = qconf->tx_mbufs[port].len;
151  qconf->tx_mbufs[port].m_table[len] = m;
152  len++;
153 
154  /* enough pkts to be sent */
155  if (unlikely(len == MAX_PKT_BURST)) {
156  send_burst(qconf, MAX_PKT_BURST, port);
157  len = 0;
158  }
159 
160  qconf->tx_mbufs[port].len = len;
161  return 0;
162 }

166 dataplane_send_single_packet(struct lcore_conf *qconf,

181 #ifdef DO_RFC_1812_CHECKS
182 static inline int
183 is_valid_ipv4_pkt(struct ipv4_hdr *pkt, uint32_t link_len)
184 {
185  /* From http://www.rfc-editor.org/rfc/rfc1812.txt section 5.2.2 */
186  /*
187  * 1. The packet length reported by the Link Layer must be large
188  * enough to hold the minimum length legal IP datagram (20 bytes).
189  */
190  if (link_len < sizeof(struct ipv4_hdr))
191  return -1;
192 
193  /* 2. The IP checksum must be correct. */
194  /* this is checked in H/W */
195 
196  /*
197  * 3. The IP version number must be 4. If the version number is not 4
198  * then the packet may be another version of IP, such as IPng or
199  * ST-II.
200  */
201  if (((pkt->version_ihl) >> 4) != 4)
202  return -3;
203  /*
204  * 4. The IP header length field must be large enough to hold the
205  * minimum length legal IP datagram (20 bytes = 5 words).
206  */
207  if ((pkt->version_ihl & 0xf) < 5)
208  return -4;
209 
210  /*
211  * 5. The IP total length field must be large enough to hold the IP
212  * datagram header, whose length is specified in the IP header length
213  * field.
214  */
215  if (rte_cpu_to_be_16(pkt->total_length) < sizeof(struct ipv4_hdr))
216  return -5;
217 
218  return 0;
219 }
220 #endif /* DO_RFC_1812_CHECKS */
