110  }

121  void push_bulk(rte_mbuf** obj_table, size_t n) override
122  {
123  int ret = rte_ring_enqueue_bulk(ring_, reinterpret_cast<void**>(obj_table), n);
124  if (ret < 0) {
125  if (ret == -EDQUOT ) {
126  /*
127  * Quota exceeded.
128  * The objects have been enqueued,
129  * but the high water mark is exceeded.
130  */
131  }
132  else if (ret == -ENOBUFS) {
133  /*
134  * Not enough room in the ring to enqueue;
135  * no object is enqueued.
136  */
137  struct rte_mbuf* pkts[n];
138  bool ret = pop_bulk(pkts, n);
139  if (ret) rte_pktmbuf_free_bulk(pkts, n);
140  push_bulk(obj_table, n);
141  } else {
142  throw slankdev::exception("rte_ring_enqueue_bulk: unknown");
143  }
144  }
145  }
146  bool pop_bulk(rte_mbuf** obj_table, size_t n) override
147  {
148  int ret = rte_ring_dequeue_bulk(ring_, reinterpret_cast<void**>(obj_table), n);
149  if (ret < 0) {
150  if (ret == -ENOENT) {
151  /*
152  * Not enough entries in the ring to dequeue,
153  * no object is dequeued.
154  */
155  }
156  return false;
157  }
158  return true;
159  }
160  virtual size_t count() const override { return rte_ring_count(ring_); }
161  virtual size_t size() const override { return ring_depth; }
162  virtual bool empty() const override { return rte_ring_empty(ring_)==1; }
163 };
